Cookie对象案例：

1.自动登录

<%@ page language="java" import="java.util.*" pageEncoding="UTF-8"%>
<!DOCTYPE HTML PUBLIC "-//W3C//DTD HTML 4.01 Transitional//EN">
<html>
  <head>
    <title>登录页面</title>
  </head>
  <body>
  <%
     String username ="";
     //读取所有cookie ，返回cookie数组
     Cookie[] cookies = request.getCookies();
     //遍历cookies数组
     if(cookies !=null){
        for(int i=0;i<cookies.length;i++){
            Cookie cookie = cookies[i];
            //根据cookie名称进行判断
            if(cookie.getName().equals("username")){
               //如果能够读取到名称为username的cookie就意味着已经登录过
               //就不再进行登录验证，之后转到欢迎页面
               response.sendRedirect("welcome.jsp");            
            }
        }
     }
   %>
     <form name="form1" method="post" action="login_action.jsp">
     <table>
        <tr>
            <td>用户名：</td><td><input type="text" name="username"></td>
        </tr>
        <tr>
            <td>密码：</td><td><input type="password" name="pwd"></td>
        </tr>
        <tr>
            <td></td><td><input type="checkbox" name="saveMessage">两周内自动登录</td>
        </tr>
        <tr>
            <td><input type="submit" value="登录"></td>
        </tr>
      </table>
     </form>
  </body>
</html>

2.欢迎界面

<%@ page language="java" pageEncoding="UTF-8"%>
<html>
    <head>
        <title>欢迎页面</title>
    </head>
    <body>
    <%
     String username = "";
     //读取所有cookie，返回cookie数组
     Cookie[] cookies = request.getCookies();
     for(int i=0;i<cookies.length;i++){
         Cookie cookie = cookies[i];
         if(cookie.getName().equals("username")){
             username = cookie.getValue();
         }
     }
     %>
        欢迎你：<%=username %>
    </body>
</html>

3.设置自动记住密码时长

<%@page import="com.news.entity.User"%>
<%@page import="com.news.dao.IUserDao"%>
<%@page import="com.news.dao.impl.UserDaoImpl"%>
<%@ page language="java" import="java.util.*" pageEncoding="UTF-8"%>
<!DOCTYPE HTML PUBLIC "-//W3C//DTD HTML 4.01 Transitional//EN">
<html>
  <head>
    <title>登录信息处理</title>
  </head>
  <body>
      <%  //获取用户输入
      String username = request.getParameter("username");
      String userpwd = request.getParameter("pwd");
      String saveMessage = request.getParameter("saveMessage");
      //对账号和密码进行判断
      IUserDao userDao = new UserDaoImpl();
      if(userDao.queryUserByUsernameAndUserpwd(new User(username,userpwd))){
         //如果用户选中了两周内自动登录，写入cookie
         if(saveMessage!=null){
            Cookie cookie = new Cookie("username",username);
            //设置cookie有效期为两周
            cookie.setMaxAge(60*60*24*14);
            //把cookie写入客户端
            response.addCookie(cookie);
         }
         response.sendRedirect("welcome.jsp");
      }else{
         response.sendRedirect("login_error.jsp");
      }
    %>
  </body>
</html>
